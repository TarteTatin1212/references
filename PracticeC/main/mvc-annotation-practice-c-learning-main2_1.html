// DDL（CREATE TABLE文） CREATE TABLE study_records ( id INT NOT NULL AUTO_INCREMENT, subject_id INT NOT NULL, studied_on DATE NOT NULL, minutes INT NOT NULL, memo VARCHAR(200), created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (id), FOREIGN KEY (subject_id) REFERENCES subjects(id) ); // ─── ポイント ─── // ・subject_id NOT NULL // = どの科目か必ず指定 // ・FOREIGN KEY で参照整合性を保証 // = subjects に存在しない id は // 入れられなくなる // ・ON DELETE は Vol.14 で扱う

FOREIGN KEY 制約を貼ることで、「subjects に id=99 がないのに、study_records に subject_id=99 を入れる」といったデータ不整合を、DBレベルで防げます。「嘘のデータが入らない」── 業務開発で必ず付けるべき安全装置です。

なぜ LocalDate？

DBの DATE 型はJava側で LocalDate に対応。Date（古い）ではなく必ず LocalDate を使う（Java 8以降の標準）。日付処理はVol.7・S7で完全攻略します。

// StudyRecord.java（新規） package com.example.todo.entity; import java.time.LocalDate; import java.time.LocalDateTime; import lombok.Data; @Data public class StudyRecord { private Integer id; private Integer subjectId; private LocalDate studiedOn; private Integer minutes; private String memo; private LocalDateTime createdAt; } // ─── ポイント ─── // ・DB側 subject_id ↔ Java側 subjectId // は application.properties の // map-underscore-to-camel-case で自動変換 // ・LocalDate は java.time パッケージ // ・@Data で全 getter/setter 自動生成

  4. Finish

@Param とは？

SQLの #{subjectId} がJavaの引数 subjectId に対応しているとMyBatisに明示するアノテーションです。これがないと「subjectId が見つかりません」エラーになります。

// StudyRecordMapper.java（新規） package com.example.todo.mapper; import java.util.List; import org.apache.ibatis.annotations.Mapper; import org.apache.ibatis.annotations.Param; import org.apache.ibatis.annotations.Select; import com.example.todo.entity.StudyRecord; @Mapper public interface StudyRecordMapper { @Select("SELECT * FROM study_records " + "WHERE subject_id = #{subjectId} " + "ORDER BY studied_on") List<StudyRecord> findBySubjectId( @Param("subjectId") Integer subjectId); } // ─── ポイント ─── // ・findBySubjectId(1) と呼べば // subject_id = 1 の記録だけ返る // ・@Param で「Javaの引数 subjectId を // SQLの #{subjectId} に渡す」と明示 // ・ORDER BY で日付順

「Java側で1対多を表現」

S1で学んだ通り、DB側では study_records に subject_id を持たせます。Java側では逆に、Subject側に「自分の記録のリスト」を持たせるのが自然です。「画面に出すときの形」に合わせる、というのがJavaオブジェクトの設計思想。

// Subject.java（改修後） package com.example.todo.entity; import java.time.LocalDateTime; import java.util.List; // ★ 追加 import lombok.Data; @Data public class Subject { private Integer id; private String name; private String color; private LocalDateTime createdAt; // ★ 追加：子のリスト private List<StudyRecord> records; } // ─── ポイント ─── // ・records は DB の列じゃない // → Service で詰めてあげる // ・@Data があるので // getRecords()/setRecords() 自動生成
modify-subject-service ④ SubjectService に findAllWithRecords()（改修）
// 手順
  1. 既存の SubjectService.java を開く
  2. StudyRecordMapper を @Autowired で注入
  3. findAllWithRecords() メソッドを追加
  4. import文を追加（Ctrl+Shift+O）
これが「2回SQL」の正体

findAll() で subjects を1回、ループ内で findBySubjectId() を subjects件数（=5回） 実行。合計1+5=6回のSQLがコンソールに流れます。これが N+1 問題の典型例。

// SubjectService.java（改修後） package com.example.todo.service; import java.util.List; import org.springframework.beans.factory.annotation.Autowired; import org.springframework.stereotype.Service; import org.springframework.transaction.annotation.Transactional; import com.example.todo.entity.Subject; import com.example.todo.entity.StudyRecord; import com.example.todo.mapper.SubjectMapper; import com.example.todo.mapper.StudyRecordMapper; @Service @Transactional public class SubjectService { @Autowired private SubjectMapper subjectMapper; @Autowired // ★ 追加 private StudyRecordMapper studyRecordMapper; public List<Subject> findAll() { return subjectMapper.findAll(); } // ★ 新規メソッド public List<Subject> findAllWithRecords() { List<Subject> subjects = subjectMapper.findAll(); // ← SQL ① for (Subject s : subjects) { List<StudyRecord> records = studyRecordMapper .findBySubjectId(s.getId()); // ← SQL ②×N s.setRecords(records); } return subjects; } }
modify-subject-controller ⑤ SubjectController を改修
// 手順
  1. 既存の SubjectController.java を開く
  2. list() メソッドの findAll() を findAllWithRecords() に差し替え

たった1行の変更です。これだけでViewに渡るデータが「Subjectだけ」から「Subject＋記録リスト」に変わります。

// SubjectController.java（改修後） package com.example.todo.controller; import org.springframework.beans.factory.annotation.Autowired; import org.springframework.stereotype.Controller; import org.springframework.ui.Model; import org.springframework.web.bind.annotation.GetMapping; import com.example.todo.service.SubjectService; @Controller public class SubjectController { @Autowired private SubjectService subjectService; @GetMapping("/subjects") public String list(Model model) { model.addAttribute("subjects", subjectService .findAllWithRecords()); // ★ 差し替え return "subjects/list"; } } // ─── ポイント ─── // ・findAll() → findAllWithRecords() // ・templates のパスは "subjects/list" のまま // ・HTML側は次で改修

08

ハマりどころと対処

TROUBLESHOOTING

Vol.2で起きやすいトラブル5選。FOREIGN KEY 関連と「N+1の違和感」がメインです。

troubles 起きやすいトラブル
// よくあるトラブル
TROUBLE 1
CREATE TABLE で Cannot add foreign key constraint
原因：参照先の subjects テーブルが存在しない／文字コード不一致／subjects.id の型と合わない。
対処：SHOW CREATE TABLE subjects; で確認。subjects.id が INT なら subject_id INT に揃える。
TROUBLE 2
INSERT で Cannot add or update a child row
原因：存在しない subject_id（例：99）を入れようとした。
対処：SELECT id FROM subjects; で存在する id を確認してINSERT。
TROUBLE 3
画面に記録が表示されない
原因：Subject に records フィールドを追加し忘れ／@Data の自動生成がエラー／Service で setRecords し忘れ。
対処：findAllWithRecords() の中で s.setRecords(records) しているか確認。
TROUBLE 4
studiedOn が null と表示される
原因：DBが studied_on／Javaが studiedOn で命名がズレ、かつ map-underscore-to-camel-case が無効。
対処：application.properties に mybatis.configuration.map-underscore-to-camel-case=true があるか確認（Aで設定済みのはず）。
TROUBLE 5
Parameter 'subjectId' not found
原因：Mapperメソッドの引数に @Param("subjectId") が付いていない。
対処：@Param("subjectId") Integer subjectId と付ける。
// FK制約エラーの典型出力 // ── TROUBLE 1 の出力 ── ERROR 1215 (HY000): Cannot add foreign key constraint // ── 原因チェックの定型 ── // ① 参照先テーブル存在？ SHOW TABLES; // ② 参照先列の型一致？ SHOW CREATE TABLE subjects; // ↑ subjects.id INT NOT NULL AUTO_INCREMENT // ↓ study_records.subject_id INT NOT NULL // ③ ENGINE / CHARSET 揃ってる？ // 通常は両方 InnoDB / utf8mb4 // ── TROUBLE 2 の出力 ── ERROR 1452 (23000): Cannot add or update a child row: a foreign key constraint fails // ── 対処：存在する id でINSERT ── SELECT id, name FROM subjects; // ↑ ここに出ている id しか使えない
